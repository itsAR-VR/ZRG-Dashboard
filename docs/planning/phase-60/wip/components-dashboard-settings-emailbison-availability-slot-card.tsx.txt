"use client";

import { useCallback, useEffect, useMemo, useState } from "react";
import { Activity, AlertTriangle, Loader2, Mail, RefreshCcw } from "lucide-react";
import { toast } from "sonner";

import {
  generateEmailBisonAvailabilitySlotPreview,
  getEmailBisonAvailabilitySlotFeatureStatus,
  runEmailBisonAvailabilitySlotDryRun,
  type EmailBisonAvailabilitySlotFeatureStatus,
} from "@/actions/emailbison-availability-slot-actions";
import { Badge } from "@/components/ui/badge";
import { Button } from "@/components/ui/button";
import { Card, CardContent, CardDescription, CardHeader, CardTitle } from "@/components/ui/card";
import { Separator } from "@/components/ui/separator";

export function EmailBisonAvailabilitySlotCard(props: {
  activeWorkspace?: string | null;
  isWorkspaceAdmin?: boolean;
}) {
  const clientId = props.activeWorkspace ?? null;
  const isWorkspaceAdmin = props.isWorkspaceAdmin ?? false;

  const [status, setStatus] = useState<EmailBisonAvailabilitySlotFeatureStatus | null>(null);
  const [statusLoading, setStatusLoading] = useState(false);
  const [previewLoading, setPreviewLoading] = useState(false);
  const [preview, setPreview] = useState<Awaited<ReturnType<typeof generateEmailBisonAvailabilitySlotPreview>>["data"] | null>(null);
  const [dryRunLoading, setDryRunLoading] = useState(false);
  const [dryRun, setDryRun] = useState<Awaited<ReturnType<typeof runEmailBisonAvailabilitySlotDryRun>>["data"] | null>(null);

  const isEmailBisonConfigured = useMemo(() => {
    if (!status) return false;
    return status.emailProvider === "EMAILBISON" && status.hasEmailBisonApiKey;
  }, [status]);

  const canRunDryRun = Boolean(clientId && isWorkspaceAdmin && isEmailBisonConfigured);

  const loadStatus = useCallback(async () => {
    if (!clientId) return;

    setStatusLoading(true);
    try {
      const res = await getEmailBisonAvailabilitySlotFeatureStatus(clientId);
      if (!res.success || !res.data) {
        setStatus(null);
        toast.error(res.error ?? "Failed to load EmailBison availability_slot status");
        return;
      }
      setStatus(res.data);
    } catch (error) {
      console.error(error);
      setStatus(null);
      toast.error("Failed to load EmailBison availability_slot status");
    } finally {
      setStatusLoading(false);
    }
  }, [clientId]);

  const loadPreview = useCallback(
    async (opts?: { refreshIfStale?: boolean }) => {
      if (!clientId) return;

      setPreviewLoading(true);
      try {
        const res = await generateEmailBisonAvailabilitySlotPreview(clientId, { refreshIfStale: opts?.refreshIfStale });
        if (!res.success) {
          toast.error(res.error ?? "Failed to generate preview");
          return;
        }
        setPreview(res.data ?? null);
      } catch (error) {
        console.error(error);
        toast.error("Failed to generate preview");
      } finally {
        setPreviewLoading(false);
      }
    },
    [clientId]
  );

  useEffect(() => {
    setStatus(null);
    setPreview(null);
    setDryRun(null);

    if (!clientId) return;
    void loadStatus();
    void loadPreview({ refreshIfStale: false });
  }, [clientId, loadPreview, loadStatus]);

  const handleDryRun = async () => {
    if (!clientId) return;
    if (!isWorkspaceAdmin) {
      toast.error("Admin access required", { description: "Only workspace admins can run provider scans." });
      return;
    }
    if (!isEmailBisonConfigured) {
      toast.error("EmailBison not configured", { description: "Set Email Provider to EmailBison and add an API key." });
      return;
    }

    setDryRunLoading(true);
    try {
      const res = await runEmailBisonAvailabilitySlotDryRun(clientId, { timeBudgetMs: 15_000 });
      if (!res.success || !res.data) {
        toast.error(res.error ?? "Dry run failed");
        return;
      }
      setDryRun(res.data);
      toast.success("Dry run complete");
    } catch (error) {
      console.error(error);
      toast.error("Dry run failed");
    } finally {
      setDryRunLoading(false);
    }
  };

  if (!clientId) return null;

  return (
    <Card>
      <CardHeader className="space-y-2">
        <div className="flex items-start justify-between gap-3">
          <div className="space-y-1">
            <CardTitle className="flex items-center gap-2">
              <Mail className="h-5 w-5" />
              EmailBison first-touch availability slots
            </CardTitle>
            <CardDescription>
              Sets the EmailBison lead custom variable{" "}
              <code className="rounded bg-muted px-1 py-0.5 text-xs">{status?.availabilitySlotVariableName ?? "availability_slot"}</code>{" "}
              just-in-time (~15 minutes before the first scheduled email).
            </CardDescription>
          </div>
          {statusLoading ? (
            <Badge variant="secondary">
              <Loader2 className="h-3 w-3 mr-1 animate-spin" />
              Loading
            </Badge>
          ) : status ? (
            isEmailBisonConfigured ? (
              <Badge className="bg-emerald-500/10 text-emerald-200 border-emerald-500/30">Active</Badge>
            ) : (
              <Badge variant="secondary">Not configured</Badge>
            )
          ) : null}
        </div>
      </CardHeader>
      <CardContent className="space-y-4">
        {status ? (
          <>
            <div className="grid grid-cols-1 gap-3 text-sm md:grid-cols-2">
              <div className="rounded border p-3">
                <div className="text-xs text-muted-foreground">Email provider</div>
                <div className="font-medium">{status.emailProvider ?? "None"}</div>
              </div>
              <div className="rounded border p-3">
                <div className="text-xs text-muted-foreground">EmailBison campaigns synced</div>
                <div className="font-medium">{status.campaignsSynced}</div>
              </div>
              <div className="rounded border p-3">
                <div className="text-xs text-muted-foreground">Workspace timezone</div>
                <div className="font-medium">{status.timezone ?? "Not set"}</div>
              </div>
              <div className="rounded border p-3">
                <div className="text-xs text-muted-foreground">Auto-book meetings</div>
                <div className="font-medium">{status.autoBookMeetings ? "On" : "Off"}</div>
              </div>
            </div>

            {status.airtableMode ? (
              <div className="flex items-start gap-2 rounded-lg border bg-muted/30 p-3 text-sm">
                <AlertTriangle className="h-4 w-4 mt-0.5 text-muted-foreground" />
                <div className="space-y-1">
                  <p className="font-medium">Airtable Mode is ON</p>
                  <p className="text-xs text-muted-foreground">
                    Email steps are typically handled externally. The <code className="rounded bg-background px-1">availability_slot</code>{" "}
                    injection may not apply unless your EmailBison templates use it.
                  </p>
                </div>
              </div>
            ) : null}
          </>
        ) : null}

        <Separator />

        <div className="space-y-2">
          <div className="flex items-center justify-between gap-2">
            <p className="text-sm font-medium flex items-center gap-2">
              <Activity className="h-4 w-4" />
              Preview
            </p>
            <Button
              variant="outline"
              size="sm"
              onClick={() => loadPreview({ refreshIfStale: true })}
              disabled={previewLoading}
            >
              {previewLoading ? (
                <>
                  <Loader2 className="h-4 w-4 mr-2 animate-spin" />
                  Refreshing…
                </>
              ) : (
                <>
                  <RefreshCcw className="h-4 w-4 mr-2" />
                  Refresh
                </>
              )}
            </Button>
          </div>

          {preview?.sentence ? (
            <div className="rounded border bg-muted/30 p-3 text-sm">
              <div className="text-xs text-muted-foreground mb-1">
                Example value for{" "}
                <code className="rounded bg-background px-1">{status?.availabilitySlotVariableName ?? "availability_slot"}</code>
              </div>
              <code className="block whitespace-pre-wrap">{preview.sentence}</code>
              {preview.availabilityError ? (
                <div className="mt-2 text-xs text-muted-foreground">Availability note: {preview.availabilityError}</div>
              ) : null}
            </div>
          ) : preview?.availabilityError ? (
            <div className="text-sm text-muted-foreground">
              No cached availability found. {preview.availabilityError}
            </div>
          ) : (
            <div className="text-sm text-muted-foreground">No preview available yet.</div>
          )}
        </div>

        <Separator />

        <div className="flex flex-wrap items-center justify-between gap-3">
          <div className="text-sm text-muted-foreground">
            Dry run scans due leads and reports what would be patched. No provider writes.
          </div>
          <Button onClick={handleDryRun} disabled={!canRunDryRun || dryRunLoading}>
            {dryRunLoading ? (
              <>
                <Loader2 className="h-4 w-4 mr-2 animate-spin" />
                Running…
              </>
            ) : (
              "Run dry run"
            )}
          </Button>
        </div>

        {!isWorkspaceAdmin ? (
          <div className="text-xs text-muted-foreground">Admin access required to run dry runs.</div>
        ) : null}

        {dryRun ? (
          <div className="grid grid-cols-2 gap-2 text-sm md:grid-cols-4">
            <div className="rounded border p-2">
              <div className="text-xs text-muted-foreground">Leads scanned</div>
              <div className="font-medium">{dryRun.leadsScanned}</div>
            </div>
            <div className="rounded border p-2">
              <div className="text-xs text-muted-foreground">Due within 15m</div>
              <div className="font-medium">{dryRun.leadsDueWithin15m}</div>
            </div>
            <div className="rounded border p-2">
              <div className="text-xs text-muted-foreground">Would patch</div>
              <div className="font-medium">{dryRun.leadsWouldUpdate}</div>
            </div>
            <div className="rounded border p-2">
              <div className="text-xs text-muted-foreground">Errors</div>
              <div className="font-medium">{dryRun.errors}</div>
            </div>
          </div>
        ) : null}
      </CardContent>
    </Card>
  );
}
